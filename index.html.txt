function getNthSaturday(year,month,n){
let date=new Date(year,month,1);
let count=0;
while(date.getMonth()===month){
if(date.getDay()===6){
count++;
if(count===n)return new Date(date.setHours(18,0,0));
}
date.setDate(date.getDate()+1);
}
}

function getNextEvent(){
const now=new Date();
const year=now.getFullYear();
const may=getNthSaturday(year,4,2);
const dec=getNthSaturday(year,11,1);
if(now<may)return may;
if(now<dec)return dec;
return getNthSaturday(year+1,4,2);
}
